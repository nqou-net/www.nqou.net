#!/usr/bin/perl

BEGIN{
    print "Content-type: text/plainnn";
    open(STDERR, ">&STDOUT");
    $|=1;
}

use Benchmark;

$count  = 1000;
$cnt    = 100;
$dat1   = "reftest1.txt";
$dat2   = "reftest2.txt";

srand;
for(0 .. $cnt){
    @nums[$_] = rand;
}

open(OUT, "> $dat1");
close(OUT);
open(OUT, "> $dat2");
close(OUT);

sleep 1;

@t = timethese($count, {
    'Refuse'  =>   '&use_ref;',
    'RefNouse'    =>   '&nouse_ref;',
});

exit(0);

sub use_ref{
    my $r_nums;
    $r_nums = @nums;
    open(OUT, ">> $dat1");
    foreach(@$r_nums){
        print OUT "$_\n";
    }
    close(OUT);
}

sub nouse_ref{
    my @n_nums;
    @n_nums = @nums;
    open(OUT, ">> $dat2");
    foreach(@n_nums){
        print OUT "$_\n";
    }
    close(OUT);
}
