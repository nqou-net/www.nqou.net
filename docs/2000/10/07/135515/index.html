sub getForm{
    my $method = $ENV{REQUEST_METHOD};
    my $len = $ENV{CONTENT_LENGTH};
    my($query, $key, $value, @elm, %form);
    if($method eq 'POST'){
        error('Memory Over') if(8191 < $len);
        read(STDIN, $query, $len);
    }elsif($method eq 'GET'){
        $query = $ENV{QUERY_STRING};
    }else{
        error('Unknown Request');
    }
    @elm = split('&amp;', $query);
    foreach(@elm){
        ($key, $value) = split('=');
        $form{$key} = $value;
    }
    return %form;
}

sub numbering{
    my($num) = @_;
    my($f);
    error('Not a Number', $num) if($num !~ /^[0-9]*.?[0-9]*$/);
    $num =~ s/^0+//;
    if($num =~ /./){
        ($num, $f) = split(/./, $num);
        $f =~ s/0+$// if $f;
    }
    $num = '0' unless $num;
    $num = divideNum($num);
    $num .= '.'.$f if $f;
    return $num;
}

sub divideNum{
    my($num) = @_;
    my $len = length($num);
    if(3 < $len){
        my $hn = substr($num, -3);
        my $th = divideNum(substr($num, 0, $len - 3));
        return $th.','.$hn;
    }else{
        return $num;
    }
}
