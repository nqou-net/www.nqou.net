package Calculator;
use strict;
use warnings;

sub new {
    my $class = shift;
    return bless {}, $class;
}

sub add {
    my ($self, $a, $b) = @_;
    return $a + $b;
}

    my ($self, $a, $b) = @_;
    die "Division by zero" if $b == 0;
    return $a / $b;
}
