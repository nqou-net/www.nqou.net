  # install Perlbrew
  curl -L http://install.perlbrew.pl | bash
  echo 'source ~/perl5/perlbrew/etc/bashrc' >> ~/.bashrc
  source ~/.bashrc

  # install Perl
  perlbrew install 5.16.3

  perlbrew switch 5.16.3

  # install App::cpanminus (cpanm)
  curl -L http://cpanmin.us | perl - App::cpanminus
fi
