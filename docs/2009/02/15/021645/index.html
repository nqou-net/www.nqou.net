package Shape::Composite;
use Moose;
use MooseX::AttributeHelpers;
with 'Shape';
has 'shapes' => (
metaclass => 'Collection::Array',
is  => 'rw',
isa => 'ArrayRef',
auto_deref => 1,
default => sub { [] },
provides => {
push => 'shape_add',
},
);
no Moose;
no MooseX::AttributeHelpers;
sub resize {
my ($self, $factor) = @_;
foreach my $shape ( $self->shapes ) {
$shape->resize( $factor );
}
}
sub print {
my $self = shift;
foreach my $shape ( $self->shapes ) {
$shape->print;
}
}
1;
