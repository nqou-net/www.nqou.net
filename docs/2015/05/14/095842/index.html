my $count = 10;
while ($count--) {
    $resque->push(
        echo => +{
            class => 'MyTask::Echo',

    );
    infof($count);

    auto_start => 0,
    tmpdir     => tempdir(),

    code => sub {
        $redis->exec;

    code   => [qw{carton exec -- perl worker.pl}],

    tag    => 'worker'
);

    code   => [qw{carton exec -- perl client.pl}],

);
