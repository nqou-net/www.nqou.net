package WWW::Mixi::Scraper::Plugin::ShowBBSPicture;
use strict;
use warnings;
use WWW::Mixi::Scraper::Plugin;
use WWW::Mixi::Scraper::Utils qw( _uri );
validator {qw(
id        is_number
comm_id   is_number
number    is_number
)};
sub scrape {
my ($self, $html) = @_;
my %scraper;
$scraper{picture} = scraper {
process 'img',
link => '@src';
result qw( link );
};
# bbs picture
my $stash;
$stash->{link} = _uri($scraper{picture}->scrape(\$html));
return $stash;
}
1;
