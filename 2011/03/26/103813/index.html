AutoInstall.pmгҒЁlib::xiгҒЁWindows

AutoInstall.pmгҒҜгҒқгҒ®гҒҫгҒҫгҒ§гҒҜеӢ•гҒҚгҒҫгҒӣгӮ“гҖӮ
lib::xiгҒ®ж–№гҒҜгҖҒcpanmгҒ§гҒҜгӮӨгғігӮ№гғҲгғјгғ«еҮәжқҘгҒҫгҒӣгӮ“гҖӮ
гҒҫгҒӮгҖҒгҒ©гҒҶгҒӣWindowsгҒӘгӮ“гҒҰгҖӮ
…гҒЁгҖҒиЁҖгҒЈгҒҰгҒ„гҒҰгӮӮд»•ж–№гҒӮгӮҠгҒҫгҒӣгӮ“гҖӮ
гҒӘгӮ“гҒЁгҒӢгҒ—гӮҲгҒҶгҒ§гҒҜгҒӮгӮҠгҒҫгҒӣгӮ“гҒӢгҖӮ
AutoInstallгҒҢеӢ•гҒӢгҒӘгҒ„гҒ®гҒҜгҖҒWindowsгҒ«гҖҢ/usr/bin/envгҖҚгҒҢз„ЎгҒ„гҒ гҒ‘гҒӘгҒ®гҒ§гҖҒгҒЎгӮҮгҒЈгҒЁжӣёгҒҚжҸӣгҒҲгӮҢгҒ°жҷ®йҖҡгҒ«дҪҝгҒҲгҒҫгҒҷгҖӮ
йҮҚе®қгҒ—гҒҰгҒ„гҒҫгҒ—гҒҹгҖӮ

package AutoInstall;

perl -MAutoInstall pl.pl

use strict;
use warnings;
BEGIN {
push @INC, sub {
my ($self, $filename) = @_;
my ($modulename) = ($filename =~ /^(.*).pm$/m);
return unless $modulename;
$modulename =~ s{/}{::}g;

system('/usr/bin/env', 'cpanm', $modulename);

system( qq{cpanm $modulename} );
my $result;
foreach my $prefix (@INC) {
ref($prefix) and next;
my $realfilename = "$prefix/$filename";
if (-f $realfilename) {
open my $fh, "<", $realfilename;
return $fh;
}
}
};
}
1;

дёҖж–№гҖҒlib::xiгҒ®ж–№гҒҜгҖҒз”ҹгҒ®гғ•гӮЎгӮӨгғ«гӮ’гҒқгҒ®гҒҫгҒҫдҪҝгҒҲгҒ°еӢ•гҒҚгҒҫгҒҷгҖӮ
з”ҹгҒ®гғ•гӮЎгӮӨгғ«гӮ’гҖҢxi.pmгҖҚгҒЁгҒ„гҒҶгҖҢеҗҚеүҚгӮ’д»ҳгҒ‘гҒҰдҝқеӯҳгҖҚгҒ—гҒҰгҖҒscript.plгҒЁеҗҢгҒҳгғ•гӮ©гғ«гғҖгҒ«гҒӮгӮӢгҖҢlibгҖҚгҒЁгҒ„гҒҶеҗҚеүҚгҒ®гғ•гӮ©гғ«гғҖгҒ®дёӯгҒ«гӮігғ”гғјгҒ—гҒҫгҒҷгҖӮ

    lib::xiгҒҜcpanmгҒ«еҜҫгҒ—гҒҰгӮӘгғ—гӮ·гғ§гғігӮ’иЁӯе®ҡгҒ§гҒҚгӮӢгҒ®гҒҢгҒ„гҒ„гҒ§гҒҷгҒӯгҖӮ
    жңҖеҲқгҒ«гғ•гӮ©гғ«гғҖеҗҚгӮ’жӣёгҒҸгҒЁгҖҒиҮӘеӢ•зҡ„гҒ«гҖҢвҖ“local-libгҖҚгӮӘгғ—гӮ·гғ§гғігҒ®гғ•гӮ©гғ«гғҖеҗҚгҒЁгҒ—гҒҰиӘҚиӯҳгҒ—гҒҰгҒҸгӮҢгҒҫгҒҷгҖӮ
    гҒӘгҒ®гҒ§гҖҒгӮ№гӮҜгғӘгғ—гғҲгҒ«д»ҘдёӢгҒ®гӮҲгҒҶгҒ«жӣёгҒ„гҒҰгҒҠгҒҸгҒЁгҖҒ
    use lib::xi qw/extlib/;

    гҒқгҒ®гӮ№гӮҜгғӘгғ—гғҲгӮ’е®ҹиЎҢгҒҷгӮӢгҒ”гҒЁгҒ«гҖҒи¶ігӮҠгҒӘгҒ„гғўгӮёгғҘгғјгғ«гҒҢгҒӮгӮҢгҒ°гҖҒгӮ№гӮҜгғӘгғ—гғҲгҒЁеҗҢгҒҳгғ•гӮ©гғ«гғҖгҒ«гҒӮгӮӢгҖҢextlibгҖҚгҒЁгҒ„гҒҶгғ•гӮ©гғ«гғҖгҒ®дёӯгҒ«гӮӨгғігӮ№гғҲгғјгғ«гҒ•гӮҢгҒҰгҒ„гҒҚгҒҫгҒҷгҖӮ
    гҒ—гҒӢгӮӮгҖҒеҲҘйҖ”гҖҢuse libгҖҚгҒ—гҒӘгҒҸгҒҰгӮӮиүҜгҒ„гҒ®гҒ§гҖҒгӮ№гӮҜгғӘгғ—гғҲгӮӮгҒҷгҒЈгҒҚгӮҠгҒ§гҒҷгҖӮ
    гҒҹгҒ гҒ—гҖҒгҒ“гҒ®е ҙеҗҲгҒ«жіЁж„ҸгҒҢеҝ…иҰҒгҒӘгҒ®гҒҜгҖҒгҖҢextlibгҖҚгӮ’гҒӮгӮүгҒӢгҒҳгӮҒдҪңгҒЈгҒҰгҒҠгҒҸеҝ…иҰҒгҒҢгҒӮгӮӢгҖҒгҒЁгҒ„гҒҶгҒ“гҒЁгҒ§гҒҷгҖӮ
    гҖҢextlibгҖҚд»ҘдёӢгҒ®гғ•гӮ©гғ«гғҖгҒҜиҮӘеӢ•зҡ„гҒ«дҪңгҒЈгҒҰгҒҸгӮҢгҒҫгҒҷгҒ®гҒ§гҖҒгҖҢextlibгҖҚгҒЁгҒ„гҒҶеҗҚеүҚгҒ®гғ•гӮ©гғ«гғҖгҒ гҒ‘гҒҜдҪңгҒЈгҒҰгҒҠгҒҚгҒҫгҒ—гӮҮгҒҶгҖӮ
