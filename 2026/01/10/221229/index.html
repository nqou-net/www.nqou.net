# PaymentChecker.pm
# Perl v5.36+, Moo

package PaymentChecker;
use v5.36;
use Moo;

# ж¬ЎгҒ®гғҒгӮ§гғғгӮ«гғјгҒёгҒ®еҸӮз…§
has 'next_handler' => (
    is        => 'rw',
    predicate => 'has_next_handler',
);

# ж¬ЎгҒ®гғҒгӮ§гғғгӮ«гғјгӮ’иЁӯе®ҡгҒ—гҒҰгҖҒгғҒгӮ§гғјгғігӮ’ж§ӢзҜү
sub set_next ($self, $handler) {
    $self->next_handler($handler);
    return $handler;  # гғҒгӮ§гғјгғіж§ӢзҜүгӮ’з¶ҡгҒ‘гӮүгӮҢгӮӢгӮҲгҒҶгҒ«
}

# еҜ©жҹ»гӮ’е®ҹиЎҢпјҲгӮөгғ–гӮҜгғ©гӮ№гҒ§гӮӘгғјгғҗгғјгғ©гӮӨгғүпјү
sub check ($self, $request) {
    # гғҮгғ•гӮ©гғ«гғҲгҒҜж¬ЎгҒ®гғҸгғігғүгғ©гҒ«е§”иӯІ
    return $self->pass_to_next($request);
}

# ж¬ЎгҒ®гғҸгғігғүгғ©гҒ«еҮҰзҗҶгӮ’е§”иӯІ
sub pass_to_next ($self, $request) {
    if ($self->has_next_handler) {
        return $self->next_handler->check($request);
    }
    # гғҒгӮ§гғјгғігҒ®жңҖеҫҢгҒ«еҲ°йҒ” = е…ЁгғҒгӮ§гғғгӮҜйҖҡйҒҺ
    return { ok => 1 };
}

# LimitChecker.pm
# Perl v5.36+, Moo

package LimitChecker;
use v5.36;
use Moo;
extends 'PaymentChecker';

has 'limit' => (
    is      => 'ro',
    default => 100_000,  # гғҮгғ•гӮ©гғ«гғҲ10дёҮеҶҶ
);

sub check ($self, $request) {
    my $amount = $request->{amount} // 0;

    if ($amount >= $self->limit) {
        return {
            ok     => 0,
            reason => sprintf('йҮ‘йЎҚгҒҢдёҠйҷҗпјҲ%dеҶҶпјүгӮ’и¶…гҒҲгҒҰгҒ„гҒҫгҒҷ', $self->limit),
        };
    }

    # OKгҒӘгӮүж¬ЎгҒ®гғҒгӮ§гғғгӮ«гғјгҒё
    return $self->pass_to_next($request);
}

# ExpiryChecker.pm
# Perl v5.36+, Moo

package ExpiryChecker;
use v5.36;
use Moo;
extends 'PaymentChecker';

sub check ($self, $request) {
    my $expiry_year  = $request->{expiry_year}  // 0;
    my $expiry_month = $request->{expiry_month} // 0;

    my ($current_year, $current_month) = (localtime)[5,4];
    $current_year  += 1900;
    $current_month += 1;

    my $is_expired = $expiry_year < $current_year ||
        ($expiry_year == $current_year && $expiry_month < $current_month);

    if ($is_expired) {
        return {
            ok     => 0,
            reason => 'гӮ«гғјгғүгҒ®жңүеҠ№жңҹйҷҗгҒҢеҲҮгӮҢгҒҰгҒ„гҒҫгҒҷ',
        };
    }

    return $self->pass_to_next($request);
}

# BlacklistChecker.pm
# Perl v5.36+, Moo

package BlacklistChecker;
use v5.36;
use Moo;
extends 'PaymentChecker';

has 'blacklist' => (
    is      => 'ro',
    default => sub { [] },
);

sub check ($self, $request) {
    my $card_number = $request->{card_number} // '';

    for my $blacklisted (@{ $self->blacklist }) {
        if ($card_number eq $blacklisted) {
            return {
                ok     => 0,
                reason => 'гҒ“гҒ®гӮ«гғјгғүгҒҜдҪҝз”ЁгҒ§гҒҚгҒҫгҒӣгӮ“',
            };
        }
    }

    return $self->pass_to_next($request);
}

# payment_chain.pl
# Perl v5.36+, Moo

use v5.36;
use utf8;
use warnings;
binmode STDOUT, ':utf8';

# дёҠиЁҳгҒ®гӮҜгғ©гӮ№гӮ’ use гҒҷгӮӢ
# пјҲе®ҹйҡӣгҒҜ lib гғҮгӮЈгғ¬гӮҜгғҲгғӘгҒ«й…ҚзҪ®гҒ—гҒҰ use lib 'lib' гҒЁгҒҷгӮӢпјү

# гғҒгӮ§гғғгӮ«гғјгӮ’дҪңжҲҗ
my $limit_checker = LimitChecker->new(limit => 100_000);
my $expiry_checker = ExpiryChecker->new;
my $blacklist_checker = BlacklistChecker->new(
    blacklist => ['4111111111111111', '5500000000000004'],
);

# гғҒгӮ§гғјгғігӮ’ж§ӢзҜү
$limit_checker
    ->set_next($expiry_checker)
    ->set_next($blacklist_checker);

# еҜ©жҹ»й–Ӣе§ӢзӮ№
my $first_checker = $limit_checker;

# жұәжёҲеҜ©жҹ»гӮ’е®ҹиЎҢ
my $result = $first_checker->check({
    amount       => 50_000,
    expiry_year  => 2028,
    expiry_month => 12,
    card_number  => '4242424242424242',
});

if ($result->{ok}) {
    say "жүҝиӘҚ: жұәжёҲеҮҰзҗҶгҒ«йҖІгҒҝгҒҫгҒҷ";
}
else {
    say "жӢ’еҗҰ: $result->{reason}";
}

# payment-check-03.pl
# гғҡгғ«гғһгғјгғҲжұәжёҲеҜ©жҹ»пјҲChain of ResponsibilityзүҲпјү
# Perl v5.36+, Moo

use v5.36;
use utf8;
use warnings;
use Moo;
binmode STDOUT, ':utf8';

# ===========================================
# еҹәеә•гӮҜгғ©гӮ№: PaymentChecker
# ===========================================
package PaymentChecker {
    use Moo;

    has 'next_handler' => (
        is        => 'rw',
        predicate => 'has_next_handler',
    );

    sub set_next ($self, $handler) {
        $self->next_handler($handler);
        return $handler;
    }

    sub check ($self, $request) {
        return $self->pass_to_next($request);
    }

    sub pass_to_next ($self, $request) {
        if ($self->has_next_handler) {
            return $self->next_handler->check($request);
        }
        return { ok => 1 };
    }
}

# ===========================================
# йҮ‘йЎҚдёҠйҷҗгғҒгӮ§гғғгӮ«гғј
# ===========================================
package LimitChecker {
    use Moo;
    extends 'PaymentChecker';

    has 'limit' => (
        is      => 'ro',
        default => 100_000,
    );

    sub check ($self, $request) {
        my $amount = $request->{amount} // 0;

        if ($amount >= $self->limit) {
            return {
                ok     => 0,
                reason => sprintf('йҮ‘йЎҚгҒҢдёҠйҷҗпјҲ%dеҶҶпјүгӮ’и¶…гҒҲгҒҰгҒ„гҒҫгҒҷ', $self->limit),
            };
        }

        return $self->pass_to_next($request);
    }
}

# ===========================================
# жңүеҠ№жңҹйҷҗгғҒгӮ§гғғгӮ«гғј
# ===========================================
package ExpiryChecker {
    use Moo;
    extends 'PaymentChecker';

    sub check ($self, $request) {
        my $expiry_year  = $request->{expiry_year}  // 0;
        my $expiry_month = $request->{expiry_month} // 0;

        my ($current_year, $current_month) = (localtime)[5,4];
        $current_year  += 1900;
        $current_month += 1;

        my $is_expired = $expiry_year < $current_year ||
            ($expiry_year == $current_year && $expiry_month < $current_month);

        if ($is_expired) {
            return {
                ok     => 0,
                reason => 'гӮ«гғјгғүгҒ®жңүеҠ№жңҹйҷҗгҒҢеҲҮгӮҢгҒҰгҒ„гҒҫгҒҷ',
            };
        }

        return $self->pass_to_next($request);
    }
}

# ===========================================
# гғ–гғ©гғғгӮҜгғӘгӮ№гғҲгғҒгӮ§гғғгӮ«гғј
# ===========================================
package BlacklistChecker {
    use Moo;
    extends 'PaymentChecker';

    has 'blacklist' => (
        is      => 'ro',
        default => sub { [] },
    );

    sub check ($self, $request) {
        my $card_number = $request->{card_number} // '';

        for my $blacklisted (@{ $self->blacklist }) {
            if ($card_number eq $blacklisted) {
                return {
                    ok     => 0,
                    reason => 'гҒ“гҒ®гӮ«гғјгғүгҒҜдҪҝз”ЁгҒ§гҒҚгҒҫгҒӣгӮ“',
                };
            }
        }

        return $self->pass_to_next($request);
    }
}

package FraudChecker {

            };
